-- Parte 2

CREATE TABLE producto (
    id_producto NUMBER PRIMARY KEY,
    nombre VARCHAR2(100) NOT NULL,
    precio NUMBER(10,2) CHECK (precio > 0),
    existencia NUMBER CHECK (existencia >= 0)
);

CREATE TABLE cliente (
    id_cliente NUMBER PRIMARY KEY,
    nombre VARCHAR2(100),
    telefono VARCHAR2(20) UNIQUE,
    fecha_registro DATE DEFAULT SYSDATE
);

CREATE TABLE venta (
    id_venta NUMBER PRIMARY KEY,
    id_cliente NUMBER,
    fecha_venta DATE DEFAULT SYSDATE,
    total NUMBER(10,2) CHECK (total >= 0),
    CONSTRAINT fk_venta_cliente
        FOREIGN KEY (id_cliente) REFERENCES cliente(id_cliente)
);

CREATE TABLE detalle_venta (
    id_detalle NUMBER PRIMARY KEY,
    id_venta NUMBER,
    id_producto NUMBER,
    cantidad NUMBER CHECK (cantidad > 0),
    precio_unitario NUMBER(10,2),
    subtotal NUMBER(10,2),
    CONSTRAINT fk_detalle_venta
        FOREIGN KEY (id_venta) REFERENCES venta(id_venta),
    CONSTRAINT fk_detalle_producto
        FOREIGN KEY (id_producto) REFERENCES producto(id_producto)
);

CREATE OR REPLACE TRIGGER trg_detalle_venta_bi
BEFORE INSERT ON detalle_venta
FOR EACH ROW
DECLARE
    v_precio PRODUCTO.precio%TYPE;
BEGIN
    SELECT precio
    INTO v_precio
    FROM producto
    WHERE id_producto = :NEW.id_producto;
    :NEW.precio_unitario := v_precio;
    :NEW.subtotal := :NEW.cantidad * v_precio;
END;

SELECT trigger_name, status
FROM user_triggers
WHERE table_name = 'DETALLE_VENTA';



-- Parte 6

CREATE OR REPLACE PROCEDURE registrar_venta (
    p_id_cliente     IN NUMBER,
    p_id_producto1   IN NUMBER,
    p_cantidad1      IN NUMBER,
    p_id_producto2   IN NUMBER,
    p_cantidad2      IN NUMBER
) AS
    v_id_venta NUMBER;
    v_total    NUMBER;
BEGIN
    SELECT seq_venta.NEXTVAL INTO v_id_venta FROM dual;

    INSERT INTO venta (id_venta, id_cliente, fecha_venta)
    VALUES (v_id_venta, p_id_cliente, SYSDATE);

    INSERT INTO detalle_venta (id_detalle, id_venta, id_producto, cantidad)
    VALUES (seq_detalle_venta.NEXTVAL, v_id_venta, p_id_producto1, p_cantidad1);

    UPDATE producto
    SET existencia = existencia - p_cantidad1
    WHERE id_producto = p_id_producto1;

    -- Detalle 2
    INSERT INTO detalle_venta (id_detalle, id_venta, id_producto, cantidad)
    VALUES (seq_detalle_venta.NEXTVAL, v_id_venta, p_id_producto2, p_cantidad2);

    UPDATE producto
    SET existencia = existencia - p_cantidad2
    WHERE id_producto = p_id_producto2;

    SELECT SUM(subtotal)
    INTO v_total
    FROM detalle_venta
    WHERE id_venta = v_id_venta;

    UPDATE venta
    SET total = v_total
    WHERE id_venta = v_id_venta;

    COMMIT;
END;
